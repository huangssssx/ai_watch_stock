```python
def _clip_str(s, max_len=400):

        return None

        return None
    return ss if len(ss) <= int(max_len) else (ss[: int(max_len)] + "…")

def _split_terms(s):
    ss = _to_str(s)
    if ss is None:
        return []
    for sep in ["；", ";", "、", "，", ",", "|", "/", " "]:
        ss = ss.replace(sep, "|")
    parts = [p.strip() for p in ss.split("|")]
    return [p for p in parts if p]

def _pick(d, path, default=None):
    cur = d
    for p in path.split("."):
        if not isinstance(cur, dict):
            return default
        cur = cur.get(p)
    return cur if cur is not None else default

def _non_empty(v):
    if v is None:
        return False
    if isinstance(v, dict):
        if "error" in v:
            return False
        if "hint" in v:
            return False
        if set(v.keys()) == {"describe"}:
            return False
        return len(v) > 0
    if isinstance(v, list):
        return len(v) > 0
    return True
```

## 中短线-震荡与反转 (KDJ/WR/BIAS)

- id：`22`

### python_code

```python

symbol = context.get("symbol", "")
if not symbol:
    df = pd.DataFrame([{"提示": "未获取到股票代码"}])
else:
    import datetime
    today = datetime.datetime.now()
    # Get enough history for 24-day MA + buffer
    start_str = (today - datetime.timedelta(days=150)).strftime("%Y%m%d")
    end_str = today.strftime("%Y%m%d")

    try:
        # Fetch daily data (qfq)
        df = ak.stock_zh_a_hist(
            symbol=symbol, period="daily", start_date=start_str, end_date=end_str, adjust="qfq"
        )
    except Exception:
        df = pd.DataFrame()

    if df is None or df.empty:
        df = pd.DataFrame([{"提示": "未获取到日线数据", "股票代码": symbol}])
    else:
        # Preprocess
        df = df.sort_values(by="日期").reset_index(drop=True)
        # Ensure numeric
        for c in ["收盘", "最高", "最低", "成交量"]:
            if c in df.columns:
                df[c] = pd.to_numeric(df[c], errors="coerce")
        
        close = df["收盘"]
        high = df["最高"]
        low = df["最低"]
        
        # 1. KDJ (9, 3, 3)
        # RSV = (Close - MinLow9) / (MaxHigh9 - MinLow9) * 100
        low_min = low.rolling(9).min()
        high_max = high.rolling(9).max()
        rsv = (close - low_min) / (high_max - low_min) * 100
        
        # Initialize K, D with 50 (standard practice or just let ewm handle it)
        # Pandas ewm with alpha=1/3 is equivalent to: Y_t = 1/3 * X_t + 2/3 * Y_{t-1}
        # This matches the common definition: K = 1/3 * RSV + 2/3 * PrevK
        df['K'] = rsv.ewm(alpha=1/3, adjust=False).mean()
        df['D'] = df['K'].ewm(alpha=1/3, adjust=False).mean()
        df['J'] = 3 * df['K'] - 2 * df['D']
        
        # 2. WR (Williams %R) - 14 days
        # WR = (MaxHigh14 - Close) / (MaxHigh14 - MinLow14) * 100
        # Note: WR is usually 0 to 100 (inverted scale often used, but here we calculate raw value)
        # Typically WR > 80 is oversold, < 20 is overbought. 
        # Wait, standard formula: (HighN - C) / (HighN - LowN) * 100.
        # So if C is close to High, WR is close to 0 (Overbought).
        # If C is close to Low, WR is close to 100 (Oversold).
        h14 = high.rolling(14).max()
        l14 = low.rolling(14).min()
        df['WR14'] = (h14 - close) / (h14 - l14) * 100
        
        # 3. BIAS (6, 12, 24)
        ma6 = close.rolling(6).mean()

        for c in ['K', 'D', 'J', 'WR14', 'BIAS6', 'BIAS12', 'BIAS24']:
```

```python
    """
    根据汇总表字段判断主力类型
    ak.stock_lhb_detail_em 接口返回的 '净买额占总成交比' 为百分比值（如 3 代表 3%）

        net_buy = float(net_buy_yi)
        ratio = float(buy_ratio) 
        
        # 机构主导：净买入 > 0 且 占比超过阈值
        if net_buy > 0 and ratio > CONFIG["institution_ratio_threshold"]:
            return "机构主导（长线锁仓）", "持有/低吸"
        
        # 游资主导：净买入 > 0 但 占比未达阈值
        elif net_buy > 0 and ratio <= CONFIG["institution_ratio_threshold"]:
            return "游资主导（短线炒作）", "警惕回调/止盈"
        
        # 主力出货：净买入为负 且 绝对值超过阈值
        elif net_buy < 0 and abs(net_buy) > CONFIG["sell_threshold_yi"]:
            return "主力出货（警惕回调）", "减仓/观望"
        
        # 其他情况归为洗盘或震荡
```

```python
import datetime
import talib

symbol = context.get("symbol", "")
if not symbol:
    df = pd.DataFrame([{"提示": "未获取到股票代码"}])
else:
    today = datetime.datetime.now()
    start_str = (today - datetime.timedelta(days=120)).strftime("%Y%m%d")
    end_str = today.strftime("%Y%m%d")

    try:
        df = ak.stock_zh_a_hist(
            symbol=symbol, period="daily", start_date=start_str, end_date=end_str, adjust="qfq"
        )
    except Exception:
        df = pd.DataFrame()

    if df.empty:
        df = pd.DataFrame([{"提示": "未获取到日线数据", "股票代码": symbol}])
    else:
        df = df.sort_values(by="日期").reset_index(drop=True)
        high = df["最高"].values
        low = df["最低"].values
        close = df["收盘"].values
        
        # 1. ATR (14)
        df["ATR"] = talib.ATR(high, low, close, timeperiod=14)
        
        # 2. Donchian Channel (20 days)
        df["High20"] = df["最高"].rolling(window=20).max()
        df["Low20"] = df["最低"].rolling(window=20).min()
        
        # 3. Volatility Ratio (ATR / Close)
        df["波动率%"] = (df["ATR"] / df["收盘"] * 100).round(2)
        
        # 4. Stop Loss Suggestion (Chandelier Exit-like: High - 3*ATR)
        # Using simple Close - 2*ATR for Long
        df["建议止损价"] = (df["收盘"] - 2 * df["ATR"]).round(2)
        
        # 5. Breakout Check (Close > Ref(High20, 1))
        # Use shift(1) for High20 to represent "previous 20 days high" excluding today if strictly following rules,
        # but here we just check if today hit the 20-day high.
        # Actually standard breakout is: Today's High > Previous 20 Days High
        prev_high20 = df["High20"].shift(1)
        df["突破20日新高"] = (df["最高"] > prev_high20).astype(int)
        
        # Format
        df["日期"] = pd.to_datetime(df["日期"]).dt.strftime("%Y-%m-%d")
        df = df.tail(5).copy()
        
        cols = [
            "日期", "收盘", "ATR", "波动率%", "建议止损价", "High20", "Low20", "突破20日新高"
        ]
        df = df[[c for c in cols if c in df.columns]]
        
        # Round
        for c in ["ATR", "High20", "Low20"]:
            if c in df.columns:
                df[c] = df[c].round(3)
```
